import numpy as np 
import matplotlib.pyplot as plt  
import seaborn as sns
import pandas as pd
rates = pd.read_csv('rates.csv')
sns.set(style="whitegrid")
f, ax = plt.subplots(figsize=(10, 8))
sns.barplot(x="Percent", y="State", linewidth=10, data=rates,color="#3498db").set_title('Top U.S. States with the Highest Percentage of Poor Mental Health Among Adults', fontname='Arial', fontsize=24)
ax.set(xlim=(38.5, 42))
sns.despine(left=True, bottom=True, top = True, right = True)
